' ─── ENUMS ────────────────────────────────────────────────
enum TipoInstrumento {
    CUERDA
    VIENTO
    PERCUSION
    TECLADO
}

' ─── CLASES MODELO ────────────────────────────────────────
abstract class Musico {
    # nombre: String
    # edad: int
    + getNombre(): String
    + getEdad(): int
    + presentarse(): String
}

class Cantante {
    - tessitura: String
    + getTessitura(): String
    + tocar(): void
    + afinar(): void
    + presentarse(): String
}

class Instrumentista {
    - instrumento: Instrumento
    - calidad: int
    + getInstrumento(): Instrumento
    + getCalidad(): int
    + tocar(): void
    + afinar(): void
    + presentarse(): String
}

class Instrumento {
    - nombre: String
    - tipo: TipoInstrumento
    - marca: String
    + getNombre(): String
    + getTipo(): TipoInstrumento
    + getMarca(): String
}

class Agrupacion {
    - nombre: String
    - genero: String
    - miembros: List<Musico>
    + agregarMiembro(m: Musico): void
    + retirarMiembro(m: Musico): void
    + getMusicos(): List<Musico>
    + getNombre(): String
    + getGenero(): String
}

class Evento {
    - lugar: String
    - fecha: LocalDate
    - entretenimiento: Agrupacion
    - aforo: int

    + registrarAgrupacion(a: Agrupacion): void
    + cancelarEvento(): void
    + getFecha(): LocalDate
    + getLugar(): String
    + getAforo(): int

}
